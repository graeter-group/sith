class PepSetter(MoleculeSetter):
    """
    The equivalent to but used specifically for proteins.

    Parameters
    ==========
    pdb_file: str or path
        path to the pdb file containing the information of the protein.
    """
    def __init__(self, pdb_file):
        self.atoms = read(pdb_file)
        self.name = pdb_file

        MoleculeSetter(self.atoms)
        indexes_aminos = self.atoms.arrays['residuenumbers']
        names_aminos = self.atoms.arrays['residuenames']
        indexes_atoms = np.arange(len(self.atoms)) + 1
        self.name_atoms_raw = self.atoms.arrays['atomtypes']

        self.atom_names = {}
        self.amino_name = {}
        self.atom_indexes = {}
        self.amino_info = {}

        for i in range(len(indexes_aminos)):
            self.amino_name[indexes_aminos[i]] = \
                names_aminos[i].replace(" ", "")
            self.atom_names[indexes_aminos[i]] = []
            self.atom_indexes[indexes_aminos[i]] = []
            self.amino_info[indexes_aminos[i]] = {}
        for i in range(len(indexes_aminos)):
            index = indexes_aminos[i]
            self.atom_names[index].append(self.name_atoms_raw[i])
            self.atom_indexes[index].append(indexes_atoms[i])
            self.amino_info[index][self.name_atoms_raw[i]] = indexes_atoms[i]

    def compute_dihedrals(self, atom1index, atom2index, atom3index,
                          atom4index, structure=None):
        """
        Compute the dihedral angle between 4 atoms. Consider the nexts facts:
        - The dihedral is defined here as the angle between the projection of
        the vector 2->1 and 3->4 in the plane generated by the normal vector
        3->2.
        - The zero of the angle is set at the projection of 2->1.
        - The positive direction of the angle is clock-wise (left-hand-rule
        respect to the normal vector) and its range is [-pi, pi].
        - The atoms do not have to be bonded.
        - vmd also meassures dihedrals as it is explained here.

        Parameters
        ==========
        atom1index: int
            index of the atom 1 in 1-based convention.
        atom2index: int
            index of the atom 2 in 1-based convention.
        atom3index: int
            index of the atom 3 in 1-based convention.
        atom4index: int
            index of the atom 4 in 1-based convention.
        structure: ase.Atoms. Default=None
            Structure to obtain the dihedrals. This has to be the same peptide
            chemistry than the basis.

        Return
        ======
        dihedral: float
            dihedral angle as defined in the description in radians.
        """
        if structure is None:
            atoms = self.atoms
        else:
            atoms = structure
        # axis of rotation of the dihedral angle
        posat1 = atoms[atom1index - 1].position
        posat2 = atoms[atom2index - 1].position
        posat3 = atoms[atom3index - 1].position
        posat4 = atoms[atom4index - 1].position

        axis = posat3 - posat2
        # vector 2->1
        axis1 = posat1 - posat2
        # vector 3->4
        axis2 = posat4 - posat3

        # projection in the plane generated by axis
        a = axis1 - axis * (round(np.dot(axis1, axis) / np.dot(axis, axis), 6))
        b = axis2 - axis * (round(np.dot(axis2, axis) / np.dot(axis, axis), 6))

        lena = np.linalg.norm(a)
        lenb = np.linalg.norm(b)
        # value of the angle
        theta = abs(np.arccos(np.dot(a, b) / (lena * lenb)))
        # sign of the angle
        cross = np.cross(a, b)
        sign = np.sign(np.dot(cross, axis))

        return sign * theta

    def compute_angle(self, atom1index, atom2index, atom3index,
                      structure=None):
        """
        Compute the angle between 3 atoms. Consider the nexts facts:

        Parameters
        ==========
        atom1index: int
            index of the atom 1 in 1-based convention.
        atom2index: int
            index of the atom 2 in 1-based convention.
        atom3index: int
            index of the atom 3 in 1-based convention.
        structure: ase.Atoms. Default=None
            Structure to obtain the dihedrals. This has to be the same peptide
            chemistry than the basis.

        Return
        ======
        (float) Angle as defined in the description in radians.

        - The index 2 should correspond to the atom in the vertex.
        - The angle is always positive and its range is [0, pi].
        - The atoms do not have to be bonded.
        """
        if structure is None:
            atoms = self.atoms
        else:
            atoms = structure
        # axis of rotation of the dihedral angle
        posat1 = atoms[atom1index - 1].position
        posat2 = atoms[atom2index - 1].position
        posat3 = atoms[atom3index - 1].position

        # vector 2->3
        a = posat3 - posat2
        # vector 2->1
        b = posat1 - posat2

        # projection in the plane generated by axis

        lena = np.linalg.norm(a)
        lenb = np.linalg.norm(b)
        # value of the angle
        theta = np.arccos(np.dot(a, b) / (lena * lenb))

        return theta

    def endo_exo_proline(self):
        """
        Computes the dihedral angles a1=(N, Ca, Cb, Cg) and a2=(Cb, Ca, N, Cd)
        in proline amino acids of the peptide. This angles meassure the
        endo-exo states, which must be given by the combination
        [a1 positive, a2 approx 0] for exo, or [a1 negative, a2 approx 0] for
        endo. 'a', 'b', 'g', and 'd' means 'aplha', 'beta', 'gamma' and 'delta'
        respectively.

        Return
        ======
        (numpy.ndarray) [#prolines x (angle1, angle2)] set of angles 1 and 2 in
        each proline.
        """
        prolines = np.where(np.array(
                            list(self.amino_name.values())) == 'PRO')[0]
        angles = np.array([[0, 0]])
        for i in prolines:
            isolated_proline = self.amino_info[i + 1]
            angle1 = self.compute_dihedrals(isolated_proline['N'],
                                            isolated_proline['CA'],
                                            isolated_proline['CB'],
                                            isolated_proline['CG'])

            angle2 = self.compute_dihedrals(isolated_proline['CB'],
                                            isolated_proline['CA'],
                                            isolated_proline['N'],
                                            isolated_proline['CD'])

            angles = np.append(angles, [[angle1, angle2]], axis=0)

        return np.delete(angles, 0, 0)

    def build_proline_state(self, aminoacid, state):
        """
        Takes a defined peptide with prolines and changes the prolines to endo
        or exo states.

        Parameters
        ==========
        aminoacid: int
            number of the aminoacid to modify. It has to correspond with a
            proline aminoacid.
        state: str
            'endo' or 'exo'. States of the proline.

        Return
        ======
        atoms: ase.Atoms object
            Atoms object with the modified structure.
        """
        # test for proline
        name_atoms = list(self.amino_info[aminoacid].keys())
        assert len(name_atoms) == 14, "wrong amino acid, check that the" + \
            f"residue number {aminoacid} is a proline and not another one."

        proline_atoms = ['N', 'CD', 'HD1', 'HD2', 'CG', 'HG1', 'HG2', 'CB',
                         'HB1', 'HB2', 'CA', 'HA', 'C', 'O']
        for atom in proline_atoms:
            assert atom in name_atoms, f"it seems that {atom} is not " + \
                f"defined in your pdb for the {aminoacid} residue."

        # defining ring atoms indexes:
        ca = self.amino_info[aminoacid]['CA']
        cb = self.amino_info[aminoacid]['CB']
        cg = self.amino_info[aminoacid]['CG']
        hg1 = self.amino_info[aminoacid]['HG1']
        hg2 = self.amino_info[aminoacid]['HG2']
        cd = self.amino_info[aminoacid]['CD']
        hd1 = self.amino_info[aminoacid]['HD1']
        hd2 = self.amino_info[aminoacid]['HD2']
        n = self.amino_info[aminoacid]['N']

        # align Ca N  CB in the same plane with Ca N in the x axis
        self.xy_alignment(ca, n, cb)
        # move Cd to the xy plane
        vec = self.atoms[cd - 1].position
        trans = self.align_plane(vec)
        self.apply_trans(trans, indexes=[cd, hd1, hd2])

        # move Cg to the xy plane
        vec = self.atoms[cg - 1].position
        trans = self.align_plane(vec)
        self.apply_trans(trans, indexes=[cg, hg1, hg2])

        # align cb, cd in the x axis, cg in the xy plane
        self.xy_alignment(cb, cd, cg)

        # endo-exo state
        if state == 'endo':
            bending = -30 * np.pi / 180
        elif state == 'exo':
            bending = 30 * np.pi / 180
        else:
            raise ValueError("The state of proline has to be either endo"
                             " or exo")

        self.apply_trans(self.rot_x(bending),
                         indexes=[cg, hg1, hg2])

        return self.atoms

    def rama_phi_psi(self, traj=None):
        """
        Compute ramachandran angles of the peptide in degrees. It assumes
        there are ACE and NME capping atoms.

        Parameters
        ==========
        traj: array-like. Default=None
            chain of ase.Atoms.

        Return
        ======
        (array) [#CA x 2float] dihedral angles per alpha carbon for
        Ramachandran plot.
        """
        if traj is None:
            traj = [self.atoms]

        angles_traj = []
        for structure in traj:
            angles = []
            if self.amino_name[1] == 'ACE':
                for i in range(2, len(self.amino_info)):
                    phi = self.compute_dihedrals(self.amino_info[i - 1]['C'],
                                                 self.amino_info[i]['N'],
                                                 self.amino_info[i]['CA'],
                                                 self.amino_info[i]['C'],
                                                 structure=structure)
                    psi = self.compute_dihedrals(self.amino_info[i + 1]['N'],
                                                 self.amino_info[i]['C'],
                                                 self.amino_info[i]['CA'],
                                                 self.amino_info[i]['N'],
                                                 structure=structure)
                    angles.append([phi * 180 / np.pi, psi * 180 / np.pi])

            elif self.amino_name[1] == 'NME':
                for i in range(2, len(self.amino_info)):
                    phi = self.compute_dihedrals(self.amino_info[i]['C'],
                                                 self.amino_info[i]['N'],
                                                 self.amino_info[i]['CA'],
                                                 self.amino_info[i + 1]['C'],
                                                 structure=structure)
                    psi = self.compute_dihedrals(self.amino_info[i - 1]['N'],
                                                 self.amino_info[i]['C'],
                                                 self.amino_info[i]['CA'],
                                                 self.amino_info[i]['N'],
                                                 structure=structure)
                    angles.append([phi * 180 / np.pi, psi * 180 / np.pi])
            angles_traj.append(angles)
        return np.array(angles_traj)
